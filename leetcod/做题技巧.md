# 常用做题技巧
**1.获取数组中的值出现的次数**
```js
//假设一个数组num=[1,2,4,5,6,7,8];统计每个数字出现的次数
let map = new Map()
num.forEach(ele => {
    let mapNum = map.get(ele)
    if(num){
        map.set(ele, ++mapNum)
    }else {
        map.set(ele, 1)
    }
})
//===========更简便一点=========
// 存在该元素就+1，不存在就置为0+1
let map = new Map()
for(let ele of num){
  map.set(ele, (map.get(ele) || 0) + 1)
}
```
