**剑指 Offer 04. 二维数组中的查找**

```javascript
/**
 * @param {number[][]} matrix
 * @param {number} target
 * @return {boolean}
 */
var findNumberIn2DArray = function(matrix, target) {
    // 查找整数是否在给定的二维数组中
    /**
    思路：
    1. 从二维数组的左下角开始查找
    2. 与目标值比较，比目标值大 上移一行
    3. 比目标值小，右移一列
    4. 找到匹配值返回 true
     */
     const [m, n] = [matrix.length, matrix[0]?.length];
     if(!m) return false
    // 定义左下角元素
    let [row, col] = [m-1, 0];
    // 循环
    while(row >= 0 && col <= n-1){
        let item = matrix[row][col];
        if(item === target) return true
        else if(item > target) row--
        else col++
    }
    return false
};
```
