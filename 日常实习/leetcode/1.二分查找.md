**剑指 Offer 04. 二维数组中的查找**

```javascript
/**
 * @param {number[][]} matrix
 * @param {number} target
 * @return {boolean}
 */
var findNumberIn2DArray = function(matrix, target) {
    // 查找整数是否在给定的二维数组中
    /**
    思路：
    1. 从二维数组的左下角开始查找
    2. 与目标值比较，比目标值大 上移一行
    3. 比目标值小，右移一列
    4. 找到匹配值返回 true
     */
     const [m, n] = [matrix.length, matrix[0]?.length];
     if(!m) return false
    // 定义左下角元素
    let [row, col] = [m-1, 0];
    // 循环
    while(row >= 0 && col <= n-1){
        let item = matrix[row][col];
        if(item === target) return true
        else if(item > target) row--
        else col++
    }
    return false
};
```

**剑指 Offer 11. 旋转数组的最小数字**
```javascript
/**
 * @param {number[]} numbers
 * @return {number}
 */
var minArray = function(numbers) {
    // 将数组开始的 几个元素 进行了一次旋转  找最小值
    // 二分查找
    /**
    三种情况：计算mid值比较
    1. numbers[mid] > numbers[right] 则 left = mid + 1
    2. numbers[mid] > numbers[right]则 right = mid
    3. numbers[mid] = numbers[right] 则 right--
     */
    let [left, right] = [0, numbers.length - 1]
    while(left <= right){
        let mid = Math.floor((left + right) / 2)
        if(numbers[mid] > numbers[right]) left = mid + 1
        else if(numbers[mid] > numbers[right]) right = mid
        else right--
    }
    return numbers[left]
};
```
**剑指 Offer 53 - I. 在排序数组中查找数字 I**

```javascript
var search = function(nums, target) {
    // 方法一暴力求解：直接从头到尾遍历一遍遇到target, 则count++
    // let count = 0;
    // nums.forEach(ele => {
    //     if(ele === target) count++
    // })
    // return count;

    // 方法二：二分法
    // 思路：找到第一个目标值，然后向两边统计
    let flag = null;
    let [left, right] = [0, nums.length-1];
    while(left <= right){
        let mid = (left + right) >> 1;
        if(nums[mid] < target) left = mid + 1;
        else if(nums[mid] > target) right = mid - 1;
        else {
            flag = mid;
            break;
        }
    }

    // 判断flag是否为空
    if(flag === null) return 0;
    left = right = flag;
    // 向两边查找
    while(nums[left-1] === target) left--
    while(nums[right+1] === target) right++
    return right - left + 1;
};
```
**剑指 Offer 53 - II. 0～n-1中缺失的数字**
```javascript
var missingNumber = function(nums) {
    // 二分查找
    /**
    思路：
    1. nums[mid] = mid; 左半边完整，则left = mid + 1;
    2. nums[mid] != mid; 左半边不完整， 则right = mid -1;
     */
     let [left, right] = [0, nums.length - 1];
     while(left <= right){
         let mid = (left + right) >> 1;
         if(nums[mid] === mid) left = mid + 1;
         else if(nums[mid] !== mid) right = mid - 1;
     }
     return left
};
```
